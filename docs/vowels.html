library(ggplot2)
vowel_plot <- ggplot(vowels, aes(x = F2, y = F1, color = vowel, label = vowel)) + labs(title = "LING2200 vowel space") +
     geom_text() + xlim(0,4000) + ylim(0,1500) + scale_y_reverse() + scale_x_reverse() +  stat_ellipse() + theme_bw()
